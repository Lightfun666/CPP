{

1.4. return initializer_list

#include <iostream>
#include <vector>
using std::cout;
using std::endl;
using std::vector;

vector<int> get1To5()
{
    return {1, 2, 3, 4, 5};
}


int main()
{
    vector<int> vec{get1To5()};

    for (auto &element : vec)
    {
        cout << element << "_";
    }
    cout << endl;
    // 1_2_3_4_5_
}

1.6. recursive

#include <iostream>
using std::cout;
using std::endl;

int factorial(int value)
{
    if (value > 1)
        return value * factorial(value - 1);
    return 1;
}

int main()
{
    cout << factorial(5) << endl; // 120
    return 0;
}

1.7. return pointer to array

#include <iostream>
using std::cout;
using std::endl;

typedef int TypeIntArr5[5];
using UseIntArr5 = int[5];

int arr[5] = {1, 2, 3, 4, 5};

// the following function return pointer to array
TypeIntArr5 *getArrayA(){return &arr;}
UseIntArr5 *getArrayB(){return &arr;}
int (*getArrayC())[5]{return &arr;}
auto getArrayD()->int(*)[5]{return &arr;}
decltype(arr) *getArrayE(){return &arr;}

int main()
{
    int (*arrPtrA)[5] = getArrayA();
    int (*arrPtrB)[5] = getArrayB();
    int (*arrPtrC)[5] = getArrayC();
    int (*arrPtrD)[5] = getArrayD();
    int (*arrPtrE)[5] = getArrayE();

    // the following are the same
    cout << *arrPtrA << endl;
    cout << *arrPtrB << endl;
    cout << *arrPtrC << endl;
    cout << *arrPtrD << endl;
    cout << *arrPtrE << endl;
    cout << &arr[0] << endl;
    return 0;
}
