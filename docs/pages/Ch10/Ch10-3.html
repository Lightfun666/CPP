1.1. predicates

sort() can use compare function of object to order

#include <vector>
#include <iostream>
#include <algorithm>

using std::cout;
using std::endl;
using std::vector;

bool comp_as_int(double i, double j);

struct myclass {
  bool operator() (int i,int j) { return (i < j);}
} myobject;

int main()
{
    double mydoubles[] = {  1.07, 2, 1.02, 1.06, -1 };
    vector<double> v;
    vector<double>::iterator it;

    v.assign(mydoubles, mydoubles + 5);
    sort(v.begin(), v.end(), comp_as_int);
    // -1 1.07 1.02 1.06  2 (v)
    //  0  1    2    3    4 (position)

    v.assign(mydoubles, mydoubles + 5);
    sort(v.begin(), v.end(), myobject);
    // -1 1.07 1.02 1.06  2 (v)
    //  0  1    2    3    4 (position)

    return 0;
}

bool comp_as_int(double i, double j)
{
    return (int(i)<int(j));
}

{

1.5. find_if()

#include <iostream>     // std::cout
#include <algorithm>    // std::find_if
#include <vector>       // std::vector

using std::cout;
using std::endl;
using std::find_if;
using std::vector;

bool IsOdd (int i) {
  return ((i%2)==1);
}

int main () {
    std::vector<int> myvector{10, 25, 40, 55};

    vector<int>::iterator it = find_if (myvector.begin(), myvector.end(), IsOdd);

    cout << "The first odd value is " << *it << '\n';

    return 0;
}

1.9. lambda implicitly capture

#include <iostream>     // std::cout

using std::cout;
using std::endl;

int main () {
    int v1 = 42 , v2 = 86;

    // capture all local variable value
    auto f0 = [=]{ cout << v1 << " " << v2 << endl; };

    // capture all local variable value, except for 'v2' reference
    auto f1 = [=, &v2]{ cout << v1 << " " << v2 << endl; };

    // capture all reference to local variables, except for 'v2' value
    auto f2 = [&, v2]{ cout << v1 << " " << v2 << endl; };

    // change local variable value
    v1 = 166;
    v2 = 199;

    f0(); // 42 86
    f1(); // 42 199
    f2(); // 166 86
    return 0;
}

1.10. mutable lambda

#include <iostream>     // std::cout

using std::cout;
using std::endl;

int main () {
    int x = 0;

    // if no 'mutable', x can not be assigned.
    // even if 'x' is assigned, the outer local varialbe 'x' is not changed
    auto lambda = [x]() mutable {
        x += 1;
        return x;
    };

    cout << lambda() << endl; // 1
    cout << x << endl; // 0

    return 0;
}

1.11. lambda return type

#include <iostream>     // std::cout

using std::cout;
using std::endl;

int main () {
    auto f0 = [](int i){ return i + 5; };

    // should add trailing return type
    auto f1 = [](int i) -> double {
        double d = i++;
        if (i > 5)
            return d + 5; // double
        else
            return i - 5; // int
    };

    cout << f0(10) << endl;
    cout << f1(3) << endl;
    return 0;
}

1.12. bind()

#include <iostream>     // std::cout
#include <string>
#include <functional>

using std::cout;
using std::endl;
using std::string;
using std::bind;
using namespace std::placeholders;

void print(const string &s1 , const string &s2);

int main () {
    auto f0 = bind(print, _1, "World!");
    auto f1 = bind(print, _2, _1);

    f0("Hi!"); // Hi! World!
    f1("world!", "Hello"); // Hello world!

    return 0;
}

void print(const string &s1, const string &s2)
{
    cout << s1 << " " << s2 << endl;
}

1.13. bind() with ref(), cref()

#include <iostream>     // std::cout
#include <functional>

using std::cout;
using std::endl;
using std::bind;
using std::ref;
using std::cref;
using namespace std::placeholders;

void print(int i1 , int &i2, const int & i3);

int main () {
    int val0 = 5;

    // use ref(), cref() to capture reference to local variable
    auto f0 = bind(print, _1, ref(val0), cref(val0));
    cout << val0 << endl; // 5

    f0(1, 2); // 1 5 5
    f0(1, 2); // 1 6 6

    cout << val0 << endl; // 7

    return 0;
}

void print(int i1 , int &i2, const int &i3)
{
    cout << i1 << " " << i2 << " " << i3 <<  endl;
    i2++;
}
