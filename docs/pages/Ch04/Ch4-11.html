1. type conversion

1.1. static_cast<>()

#include <iostream>

using std::cout;
using std::endl;

int main() {
    // sample 1: double to int
    double d = 3.14;
    int i = static_cast<int>(d);
    cout << "Double to int: " << i << endl;

    // sample 2: pointers
    int x = 10;
    void* ptr = &x;
    int* ptrToInt = static_cast<int*>(ptr);
    cout << "Pointer to int: " << *ptrToInt << endl;

    // sample 3: class heirachy
    class Base {
    public:
        virtual void foo() {}
    };
    class Derived : public Base {};
    Base* basePtr = new Derived();
    Derived* derivedPtr = static_cast<Derived*>(basePtr);

    // sample 4: enum
    enum class Color { Red, Green, Blue };
    int enumValue = static_cast<int>(Color::Green);
    cout << "Enum to int: " << enumValue << endl;

    return 0;
}

1.2. const_cast<>()

#include <iostream>
using std::cout;
using std::endl;

int main() {

    int i = 3;                 // i is not declared const
    const int& rci = i;
    const_cast<int&>(rci) = 4; // OK: modifies i
    cout << "i = " << i << '\n'; // i = 4

    const int j = 3; // j is declared const
    int* pj = const_cast<int*>(&j);
    *pj = 4;      // undefined behavior
    cout << j << endl;
    return 0;
}

1.3. reinterpret_cast<>()

#include <iostream>

int main() {
    int x = 42;
    int* intPtr = &x;

    // int* to double*, and they are irrelevant
    double* doublePtr = reinterpret_cast<double*>(intPtr);
    *doublePtr = 3.14159;

    std::cout << "x = " << x << std::endl;  

    return 0;
}

1.4. old-style casts

#include <iostream>

int main() {

    // 2 kinds of old style conversion:
    // type(expression);
    // (type)expression;

    int x = 10;
    double y = (double)x; // C style conversion
    double y = double(x); // function like
    void* voidPtr = (void*)&x; // int* to void*

    return 0;
}
