## assign, compound operators
```cpp
#include <iostream>
#include <vector>
using std::cout;
using std::endl;
using std::vector;

int main() {
    
    // these two are initialization, not assignment
    int i = 0, j = 0, k = 0;
    const int ci = i;
    
    k = 3.14; // 3
    
    // error : no narrow conversion for list initialization
    // k = {3.14};
    
    vector<int> vi;
    vi = {1, 2, 3, 4, 5};
    
    // righ associative
    i = j = k = 5;
     
    k += 5; // k= k + 5, 10
    k -= 4; // k= k - 4, 6
    k *= 2; // k= k * 2, 12
    k /= 4; // k= k / 4, 3
    return 0;
}
```
