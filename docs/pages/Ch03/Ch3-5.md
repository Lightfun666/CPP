# primitive built-in type

## type conversions
```cpp
#include<iostream>
using std::cout;
using std::endl;
int main(){
    bool bb = 4;            // b  <- true
    int ii = bb;            // i  <- 1
    ii = 3.14;              // i  <- 3
    double pi = ii;         // pi <- 3.0
    unsigned char c0= -1;   // c0  <- 255
    signed char c1 = 256;   // c1 undefined
    return 0;
}

```
