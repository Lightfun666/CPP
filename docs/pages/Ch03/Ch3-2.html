1. std::string

1.1. define and initialization

#include <string>
using std::string;
int main() {
    // default initialization, str1 is empty string
    string str1;

    // copy initialization
    string str2 = "hiya";
    string str3 = string(10, 'c');

    // direct initialization
    string str4("hiya");
    string str5(10, 'c');

    // copy
    str5 = str1;
    return 0;
}

1.2. read string

#include <iostream>
#include <string>
using std::string;
using std::cin;
using std::cout;
using std::endl;
int main() {
    // 1. cin one string
    string s;
    cin >> s;
    cout << s << endl;

    // 2. cin several string
    string s1, s2;
    cin >> s1 >> s2;
    cout << s1 << s2 << endl;

    // 3. countinous cin string
    string word;
    while(cin >> word)
        cout << word << endl;

    // 4. getline can get space but no newline
    string line;
    while(getline(cin, line))
        cout << line << endl;
    return 0;
}

1.3. size(), size_type, []

#include <iostream>
#include <string>
using std::string;
using std::cin;
using std::cout;
using std::endl;
int main() {    
    // size(), size_type
    string cppstr = "C++ Programming";
    for (string::size_type i = 0; i < cppstr.size(); ++i) {
        // []
        cout << "char " << i << ": " << cppstr[i] << endl;
    }
    return 0;
}

1.4. empty()

#include <iostream>
#include <string>
using std::string;
using std::cin;
using std::cout;
using std::endl;
int main() {    
    std::string str1 = "Hello";
    std::string str2 = "";

    if (str1.empty()) {
        std::cout << "str1 is empty" << std::endl;
    } else {
        std::cout << "str1 is not empty" << std::endl;
    }

    if (str2.empty()) {
        std::cout << "str2 is empty" << std::endl;
    } else {
        std::cout << "str2 is not empty" << std::endl;
    }
    return 0;
}

1.6. for

#include <iostream>
#include <string>
#include <cctype>
// cctype define functions of character
using std::string;
using std::cout;
using std::endl;
int main() {
    string s1 = "hello world";
    for (auto c : s1)
    {
        c = toupper(c); // not change s1
        cout << c;
    }
    cout << endl;
    cout << s1 << endl << endl;

    for (auto &c : s1) // use auto &
    {
        c = toupper(c); // change char in s1
        cout << c;
    }
    cout << endl;
    cout << s1 << endl;
    return 0;
}
