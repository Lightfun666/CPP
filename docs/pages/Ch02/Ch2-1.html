1. primitive built-in type

1.1. type conversions

#include<iostream>
using std::cout;
using std::endl;
int main(){
    bool bb = 4;            // b  <- true
    int ii = bb;            // i  <- 1
    ii = 3.14;              // i  <- 3
    double pi = ii;         // pi <- 3.0
    unsigned char c0= -1;   // c0  <- 255
    signed char c1 = 256;   // c1 undefined
    return 0;
}

#include<iostream>
using std::cout;
using std::endl;
int main(){
    int i = 4;
    if(i) // integer 4 is true
        i =2;
    cout << i << endl; // 2

    unsigned ui = 1;
    int j = -4;
    cout << j + j << endl;  // -8
    cout << ui + j << endl; // 4294967293, j transform into unsigned
    return 0;
}

1.2. literals

#include<iostream>
using std::cout;
using std::endl;
int main(){
    cout << "decimal, octal, hexadecimal" << endl;
    cout << 20 << endl;     // 20
    cout << 024 << endl;    // 20
    cout << 0x14 << endl;   // 20
    cout << endl;

    cout << "decimal point" << endl;
    cout << 3.1415 << endl;
    cout << 3.1415E0 << endl;
    cout << 0. << endl;
    cout << 0e0<< endl;
    cout << .66 << endl;
    cout << endl;

    cout << "char, string, multiline" << endl;
    cout << 'a' << endl;
    cout << "hello world !!!" << endl;
    cout << "line 1, "
            "line 2" << endl; // line 1, line2
    return 0;
}

#include<iostream>
using std::cout;
using std::endl;
int main(){
    unsigned long long ull = 42ULL;
    float f = 1E-3F;
    long double ld = 3.1415L;
    return 0;
}
