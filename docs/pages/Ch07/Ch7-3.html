1. other feature

1.1. mutable

#include <iostream>
using std::cout;
using std::endl;

class Point
{
    mutable int x;
    int y;
public:
    Point():x(5), y(6){}
    void addX() const{ x++; }
    void addY() { y++; }
    void show() const{cout << x << ' ' << y << endl;}
};

int main()
{
    Point p0;
    p0.addX();
    p0.addY();
    p0.show();

    const Point p1;
    // const object call const member function
    // that can change mutable member
    p1.addX();
    p1.show();
    return 0;
}

1.3. return *this

#include <iostream>
using std::cout;
using std::endl;

class Point
{
    int x = 0, y = 0;
public:
    Point &moveX(int delta) { x += delta; return *this;}
    Point &moveY(int delta) { y += delta; return *this;}

    void show() const{cout << x << ' ' << y << endl;}
};

int main()
{
    Point p0;
    p0.moveX(3).moveY(-2);
    p0.show(); // 3 -2

    return 0;
}

1.4. const member function return *this

#include <iostream>
using std::cout;
using std::endl;

class Point
{
    int x = 0, y = 0;
public:
    Point &moveX(int delta) { x += delta; return *this;}
    Point &moveY(int delta) { y += delta; return *this;}

    // in const member function, 'this' points to const itself.
    // hence *this is a const, and only can be referenced by const reference
    const Point &show() const{cout << x << ' ' << y << endl; return *this;}
};

int main()
{
    Point p0;
    p0.moveX(3).moveY(-2);
    p0.show();

    const Point p1;
    p1.show().show();

    return 0;
}

1.5. const overloaded

#include <iostream>
using std::cout;
using std::endl;

class Point
{
    int x = 0, y = 0;
public:
    Point(int x, int y):x(x), y(y){}
    void show() const{cout << "const: " << x << ' ' << y << endl;}
    void show(){cout << "non-const: " << x << ' ' << y << endl;}
};

int main()
{
    Point p0(3, 4);
    p0.show(); // non-const: 3 4

    const Point p1(3, 4);
    p1.show(); // const: 3 4
    return 0;
}

1.7. friend class

#include <iostream>
#include <vector>
using std::cout;
using std::endl;
using std::vector;

class Point
{
    friend class Graph; // Graph is frend, can use 'x' and 'y'
    int x = 0, y = 0;
public:
    Point(int x, int y):x(x), y(y){}
    void show() const{cout << x << ' ' << y << endl;}
};

class Graph{
    vector<Point> points;
public:
    Graph(vector<Point> ps):points(ps){}
    void resetAll()
    {
        for (auto &p : points)
            p.x = p.y = 0; // use private member of Pointer
    }
    void show()
    {
        for (const auto &p : points)
            p.show();
    }
};

int main()
{
    vector<Point> vec = {Point(1, 2), Point(2, 3)};
    Graph g(vec);
    g.resetAll();
    g.show();
    return 0;
}

1.8. friend method

#include <iostream>
#include <vector>
using std::cout;
using std::endl;
using std::vector;

class Point; // Point declare for Graph

// declare Graph
class Graph{
    vector<Point> points;
public:
    Graph(vector<Point> pts);
    void resetAll();
};

// declare Point
class Point{
    // friend a Graph member function
    friend void Graph::resetAll();
    int x, y;
public:
    Point(int x, int y);
};

Point::Point(int x, int y):x(x), y(y){}

Graph::Graph(vector<Point> pts):points(pts){}

void Graph::resetAll()
{
    // Graph member use Point private member
    for (auto &p : points)
        p.x = p.y = 0;
}

int main()
{
    cout << "hello world" << endl;
    vector<Point> vec = {Point(1,2), Point(3,4)};
    Graph g(vec);
    g.resetAll();
    return 0;
}

1.9. friend function declare and define

#include <iostream>
using std::cout;
using std::endl;

class Point{
    // friend function define here
    friend void show(Point p){ cout << p.x << " " << p.y << endl; } 
    int x, y;
public:
    Point(int x, int y):x(x), y(y){};
    void showXY(){ show(*this); }
};

int main()
{
    cout << "hello world" << endl;
    Point p(3, 4);
    p.showXY(); // 3 4
    return 0;
}
