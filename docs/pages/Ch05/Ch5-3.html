1. conditional

1.1. if

#include <iostream>
using std::cout;
using std::endl;

int main() {
    int number = 3;

    if (number > 0) {
        cout << "The number is positive." << endl;
    } else if (number < 0) {
        cout << "The number is negative." << endl;
    } else {
        cout << "The number is zero." << endl;
    }

    return 0;
}

1.2. nested if

#include <iostream>
using std::cout;
using std::endl;

int main() {
    int age = 19;
    char gender = 'F';
    if (age >= 18) {
        if (gender == 'M' || gender == 'm') {
            cout << "You are an adult male." << endl;
        } else if (gender == 'F' || gender == 'f') {
            // print this line
            cout << "You are an adult female." << endl; 
        }
    } else {
        cout << "You are not yet an adult." << endl;
    }
    return 0;
}

1.3. dangling else

#include <iostream>
using std::cout;
using std::endl;

int main() {
    int x = 5;
    if (x > 0)
        if (x < 10)
            cout << "x is greater than 0 and less than 10." << endl;
        else // x < 0 would not go here
            cout << "x is not less than 10." << endl;
    return 0;
}

1.4. switch

#include <iostream>
using std::cout;
using std::endl;
int main() {
    int choice = 3;
    switch (choice) {
        case 1:
            cout << "Hello!" << endl;
            break;
        case 2:
            cout << "Goodbye!" << endl;
            break;
        case 3:
            cout << "Quitting the program..." << endl;
            break;
        default: // make other developers know that this is considered
            cout << "Invalid choice. Please select a valid option." << endl;
            break;
    }

    return 0;
}
